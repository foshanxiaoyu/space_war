// C Multiplication Table printf()
#include <stdlib.h>
#include <stdio.h>
int main(){
    for (int x = 1; x < 10; x++)
    {
        printf("\n");
        for (int y = 1; y < x+1; y++)
        {
           printf("%2d*%2d=%2d  ",x,y,x*y);
        }
    }
    
    return 0;
}
// End
